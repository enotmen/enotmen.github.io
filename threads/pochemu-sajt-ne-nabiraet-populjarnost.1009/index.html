Открыть: /lib/CP_structure.js
Найти код:
JavaScript: 
if (movie.description_short) {
    text = movie.description_short
        .replace(/<\/?[^>]+>/g, '')
        .replace(/\s+/g, ' ')
        .replace(/(^\s*)|(\s*)$/g, '')
        .substr(0, 200)
        .replace(/"([^"]*?)"/gi, '«$1»')
        .replace('"', '&quot;');
}
Изменить:
JavaScript: 
if (movie.description_short) {
    text = (movie.description || movie.description_short)
        .replace(/<\/?[^>]+>/g, '')
        .replace(/\s+/g, ' ')
        .replace(/(^\s*)|(\s*)$/g, '')
        .substr(0, 200)
        .replace(/"([^"]*?)"/gi, '«$1»')
        .replace('"', '&quot;');
}
