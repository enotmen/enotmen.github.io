в файле /lib/CP_structure.js изменить функцию toDate.



JavaScript: 
function toDate(days) {
    var d = (days)
        ? new Date((days - 719527) * 1000 * 60 * 60 * 24)
            .toJSON().substr(0, 10)
        : '';
    var monthNames = [
        "Января", "Февраля", "Марта",
        "Апреля", "Мая", "Июня", "Июля",
        "Августа", "Сентября", "Октября",
        "Ноября", "Декабря"
    ];
    if (d) {
        return d[8] + d[9] + ' ' +
            monthNames[parseInt(d[5] + d[6]) - 1] + ' ' +
            d[0] + d[1] + d[2] + d[3];
    } else {
        return '';
    }
}
