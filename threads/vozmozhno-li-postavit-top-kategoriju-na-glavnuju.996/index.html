Открыть /routes/paths/index.js
Перед "slider": function (callback) { добавить
JavaScript: 
"top": function (callback) {
    return (modules.top.status)
        ? CP_get.additional(
        {"genre": "комедия"},
        'top',
        options,
        function (err, movies) {
            if (err) return callback(err);

            return (movies && movies.length)
                ? callback(null, movies)
            : callback(null, [])
        })
    : callback(null, [])
},
"type": "фильмы" - Топ из типа
"year": "2019" - Топ из года
"genre": "комедия" - Топ из жанра
"country": "США" - Топ из страны
"actor": "Том Круз" - Топ из актера
"director": "Джеймс Кэмерон" - Топ из режиссера
